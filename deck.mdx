interface Matcher<T, R> {
  idle: () => R
  loading: () => R
  success: (data: T) => R
  failure: (error: Error) => R
}

const match = <T, R>(m: Matcher<T, R>, ad: AsyncData<T>): R => {
  if (ad.status === 'idle') {
    return m.idle()
  }

  if (ad.status === 'loading') {
    return m.loading()
  }

  if (ad.status === 'failure') {
    return m.success(ad.error)
  }

  return m.success(ad.data)
}
